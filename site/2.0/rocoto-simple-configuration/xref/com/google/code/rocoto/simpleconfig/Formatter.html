16  package com.google.code.rocoto.simpleconfig;
17  
18  import java.util.ArrayList;
19  import java.util.List;
20  import java.util.Map;
21  
22  /**
23   * 
24   *
25   * @author Simone Tripodi
26   * @version $Id$
27   */
28  final class Formatter {
29  
30      private static final String VAR_BEGIN = "$";
31  
32      private final List<Appender> appenders = new ArrayList<Appender>();
33  
34      private boolean containsKeys = false;
35  
36      public Formatter(final String pattern) {
37          int prev = 0;
38          int pos;
39          while ((pos = pattern.indexOf(VAR_BEGIN, prev)) >= 0) {
40              if (pos > 0) {
41                  this.appenders.add(new TextAppender(pattern.substring(prev, pos)));
42              }
43              if (pos == pattern.length() - 1) {
44                  this.appenders.add(new TextAppender(VAR_BEGIN));
45                  prev = pos + 1;
46              } else if (pattern.charAt(pos + 1) != '{') {
47                  if (pattern.charAt(pos + 1) == '$') {
48                      this.appenders.add(new TextAppender(VAR_BEGIN));
49                      prev = pos + 2;
50                  } else {
51                      this.appenders.add(new TextAppender(pattern.substring(pos, pos + 2)));
52                      prev = pos + 2;
53                  }
54              } else {
55                  int endName = pattern.indexOf('}', pos);
56                  if (endName < 0) {
57                      throw new IllegalArgumentException("Syntax error in property: " + pattern);
58                  }
59                  String key = pattern.substring(pos + 2, endName);
60                  this.appenders.add(new KeyAppender(key));
61                  prev = endName + 1;
62                  this.containsKeys = true;
63              }
64          }
65          if (prev < pattern.length()) {
66              this.appenders.add(new TextAppender(pattern.substring(prev)));
67          }
68      }
69  
70      public boolean containsKeys() {
71          return this.containsKeys;
72      }
73  
74      public String format(Map<String, String> configuration) {
75          StringBuilder buffer = new StringBuilder();
76          for (Appender appender : this.appenders) {
77              appender.append(buffer, configuration);
78          }
79          return buffer.toString();
80      }
81  
82      @Override
83      public String toString() {
84          return this.appenders.toString();
85      }
86  
87  }
