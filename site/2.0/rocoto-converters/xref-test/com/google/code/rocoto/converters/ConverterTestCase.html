75      @BeforeClass
76      protected final void init() {
77          Injector injector = Guice.createInjector(new ConvertersModule(), new AbstractModule() {
78              @Override
79              protected void configure() {
80                  this.bindConstant()
81                      .annotatedWith(Names.named("bitset"))
82                      .to("a, 123, ~");
83                  this.bindConstant()
84                      .annotatedWith(Names.named("charset"))
85                      .to("UTF-8");
86                  this.bindConstant()
87                      .annotatedWith(Names.named("file"))
88                      .to("/tmp");
89                  this.bindConstant()
90                      .annotatedWith(Names.named("locale"))
91                      .to("en_US");
92                  this.bindConstant()
93                      .annotatedWith(Names.named("properties"))
94                      .to("useUnicode=true\ncharacterEncoding=UTF-8");
95                  this.bindConstant()
96                      .annotatedWith(Names.named("classpathResource"))
97                      .to("classpath:///testng.xml");
98              }
99          });
100         injector.injectMembers(this);
101     }

103     @Test
