45  /**
46   * Allows user to easily register converters not included in the base
47   * google-guice package.
48   *
49   * @author Simone Tripodi
50   * @version $Id$
51   */
52  public final class ConvertersModule extends AbstractModule {
53  
54      /**
55       * Maintains all the auxiliary converters.
56       */
57      private final Map<Matcher<? super TypeLiteral<?>>, TypeConverter> converters =
58          new HashMap<Matcher<? super TypeLiteral<?>>, TypeConverter>();
59  
60      /**
61       * Builds a new converters with default converters.
62       */
63      public ConvertersModule() {
64          this.registerConverter(BitSet.class, new BitSetConverter());
65          this.registerConverter(Charset.class, new CharsetConverter());
66          this.registerConverter(Currency.class, new CurrencyConverter());
67          this.registerConverter(File.class, new FileConverter());
68          this.registerConverter(Locale.class, new LocaleConverter());
69          this.registerConverter(Pattern.class, new PatternConverter());
70          this.registerConverter(Properties.class, new PropertiesConverter());
71          this.registerConverter(TimeZone.class, new TimeZoneConverter());
72          this.registerConverter(URL.class, new URLConverter());
73          this.registerConverter(URI.class, new URIConverter());
74          this.registerConverter(UUID.class, new UUIDConverter());
75  
76          NumberConverter numberConverter = new NumberConverter();
77          this.registerConverter(BigDecimal.class, numberConverter);
78          this.registerConverter(BigInteger.class, numberConverter);
79  
80          DateConverter dateConverter = new DateConverter();
81          this.registerConverter(Calendar.class, dateConverter);
82          this.registerConverter(Date.class, dateConverter);
83  
84          SQLDateTimeConverter sqlDateTimeConverter = new SQLDateTimeConverter();
85          this.registerConverter(java.sql.Date.class, sqlDateTimeConverter);
86          this.registerConverter(Time.class, sqlDateTimeConverter);
87          this.registerConverter(Timestamp.class, sqlDateTimeConverter);
88      }

90      /**
91       * Associates the specified converter with the specified type in this module.
92       *
93       * @param type type with which the specified converter is to be associated.
94       * @param typeConverter converter to be associated with the specified type.
95       */
96      public void registerConverter(Class<?> type, TypeConverter typeConverter) {
97          if (type == null) {
98              throw new IllegalArgumentException("Argument 'type' nust not be null");
99          }
100         if (typeConverter == null) {
101             throw new IllegalArgumentException("Argument 'typeConverter' nust not be null");
102         }
103         this.registerConverter(TypeLiteral.get(type), typeConverter);
104     }

106     /**
107      * Associates the specified converter with the specified type literal in
108      * this module.
109      *
110      * @param typeLiteral type literal with which the specified converter is to
111      *        be associated.
112      * @param typeConverter converter to be associated with the specified type.
113      */
114     public void registerConverter(TypeLiteral<?> typeLiteral, TypeConverter typeConverter) {
115         if (typeLiteral == null) {

117         }

128      * @param matcher matcher with which the specified converter is to be
129      *        associated.
130      * @param typeConverter converter to be associated with the matcher.
131      */
132     public void registerConverter(Matcher<? super TypeLiteral<?>> matcher, TypeConverter typeConverter) {
133         if (matcher == null) {
134             throw new IllegalArgumentException("Argument 'matcher' nust not be null");
135         }
136         if (typeConverter == null) {
137             throw new IllegalArgumentException("Argument 'typeConverter' nust not be null");
138         }
139         this.converters.put(matcher, typeConverter);
140     }

142     /**
143      * Returns the converter to which the specified type is mapped, or
144      * {@code null} if this module contains no mapping for the type.
145      *
146      * @param type the type whose associated converter is to be returned.
147      * @param converterType the converter type is looking for associated to the
148      *        type.
149      * @return the converter to which the specified type is mapped, or null if
150      *         this module contains no mapping for the type.
151      */
152     public <T extends TypeConverter> T lookup(Class<?> type, Class<T> converterType) {
153         if (type == null) {
154             throw new IllegalArgumentException("Argument 'type' nust not be null");
155         }
156         return this.lookup(TypeLiteral.get(type), converterType);
157     }

159     /**
160      * Returns the converter to which the specified type literal is mapped, or
161      * {@code null} if this module contains no mapping for the type.
162      *
163      * @param typeLiteral the type literal whose associated converter
164      *        is to be returned.
165      * @param converterType the converter type is looking for associated to the
166      *        type literal.
167      * @return the converter to which the specified type literal is mapped, or
168      *         null if this module contains no mapping for the type.
169      */
170     public <T extends TypeConverter> T lookup(TypeLiteral<?> typeLiteral, Class<T> converterType) {
171         if (typeLiteral == null) {
172             throw new IllegalArgumentException("Argument 'typeLiteral' nust not be null");
173         }
174         return this.lookup(Matchers.only(typeLiteral), converterType);
175     }
176 
177     /**
178      * Returns the converter to which the specified matcher is mapped, or
179      * {@code null} if this module contains no mapping for the matcher.
180      *
181      * @param matcher the matcher whose associated converter is to be returned.
182      * @param converterType the converter type is looking for associated to the
183      *        input matcher.
184      * @return the converter to which the specified matcher is mapped, or
185      *         null if this module contains no mapping for the matcher.
186      */
187     public <T extends TypeConverter> T lookup(Matcher<? super TypeLiteral<?>> matcher, Class<T> converterType) {
188         if (matcher == null) {
189             throw new IllegalArgumentException("Argument 'matcher' nust not be null");
190         }
191         TypeConverter converter = this.converters.get(matcher);
192         if (converter != null && converterType.isInstance(converter)) {
193             return converterType.cast(converter);
194         }
195         return null;
196     }
197 
198     /**
199      * {@inheritDoc}
200      */
201     @Override
202     protected void configure() {
203         for (Entry<Matcher<? super TypeLiteral<?>>, TypeConverter> converter : this.converters.entrySet()) {
204             this.binder().convertToTypes(converter.getKey(), converter.getValue());
205         }
206     }
