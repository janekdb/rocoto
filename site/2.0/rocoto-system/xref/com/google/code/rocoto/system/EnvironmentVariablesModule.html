16  package com.google.code.rocoto.system;
17  
18  import java.util.Map.Entry;
19  
20  import com.google.inject.AbstractModule;
21  
22  /**
23   * Binds Environment Variables into Guice binder.
24   *
25   * @author Simone Tripodi
26   * @version $Id$
27   */
28  public final class EnvironmentVariablesModule extends AbstractModule {
29  
30      /**
31       * {@inheritDoc}
32       */
33      @Override
34      protected void configure() {
35          for (Entry<String, String> envVar : System.getenv().entrySet()) {
36              this.bindConstant()
37                  .annotatedWith(new EnvironmentVariableImpl(envVar.getKey()))
38                  .to(envVar.getValue());
39          }
40      }
41  
42  }
